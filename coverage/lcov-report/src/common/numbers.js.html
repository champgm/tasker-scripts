const lastFourToNameMap = Object.keys(numbers)
  .reduce((collector, key) => {
    const number = numbers[key];
    const lastFour = number.substr(number.length - 4);
    collector[lastFour] = names[key];

function getNameByNumber(number) {
  const matchingFour = Object.keys(lastFourToNameMap).find((knownLastFour) => {
    const lastFour = number.substr(number.length - 4);
    return lastFour === knownLastFour;
  });
  return lastFourToNameMap[matchingFour];
}

  console.log(`contact name: ${contactName}`);
  const lowercaseContactName = contactName.toLocaleLowerCase();
  const carrier = Object.keys(carrierStageHostTypeMap).find((carrier) => {
    return lowercaseContactName.includes(carrier);
  });
 
  const stageHostMap = carrierStageHostTypeMap[carrier];
  console.log(`stage host map: ${JSON.stringify(stageHostMap, null, 2)}`);
  const stage = Object.keys(stageHostMap).find((stage) => {
    return lowercaseContactName.includes(stage);
  });
 
  const host = stageHostMap[stage];
  console.log(`got host: ${host}`);
  if (typeof host === 'string' || host instanceof String) {
    return host;
  }
 
  const type = Object.keys(host).find((type) => {
    return lowercaseContactName.includes(type);

}
