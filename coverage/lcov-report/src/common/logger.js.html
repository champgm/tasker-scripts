const numbers = require('./numbers').numbers;
 
const postBodyLoggerUrl = 'https://w084m9sdkk.execute-api.us-east-2.amazonaws.com/dev/log';
const defaultSmsLogsRecipient = numbers.googleVoiceMacWork;
const keepInMessagingApp = true;
 
function sms(context, message) {
  context.msg = message;
  const contextString = JSON.stringify(context, null, 2);
  sendSMS(defaultSmsLogsRecipient, contextString, keepInMessagingApp);
};
 
function post(context, message) {
  context.msg = message;
  const body = JSON.stringify(context);
  const requestParams = { body, method: 'post' };
  return fetch(postBodyLoggerUrl, requestParams);
};
 
module.exports.sms = sms;
module.exports.post = post;
