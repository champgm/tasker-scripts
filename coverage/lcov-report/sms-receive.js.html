const sms = require('./common/logger').sms;
const post = require('./common/logger').post;
 
const stagingTwilio = 'Staging - Twilio';
const productionTwilio = 'Production - Twilio';
const developTwilio = 'Develop - Twilio';
const macClx = 'Mac - CLX';
 
const numbersMap = {
  '0956': macClx,
  '5698': developTwilio,
  '5569': stagingTwilio,
  '77577': productionTwilio
}
const numbersMapKeys = Object.keys(numbersMap);
 
const stageMap = {
  MAC: macClx,
  DEV: developTwilio,
  STAG: stagingTwilio,
  PROD: productionTwilio
}
const stageMapKeys = Object.keys(stageMap);
 
const connectorsMap = {
  'TWILIO': 'Twilio',
  'CLX': 'CLX',
}
const connectorsMapKeys = Object.keys(connectorsMap);
 
function getMaven(notificationTitleInput) {
  const notificationTitle = ('' + notificationTitleInput).toLocaleUpperCase();
 
  // Check by number first
  for (let index = 0; index < stageMapKeys.length; index++) {
    const currentKey = stageMapKeys[index];
    if (notificationTitle.indexOf(currentKey) > -1) {
      return stageMap[currentKey];
    }
  }
 
  // Check for specified stage
  for (let index = 0; index < numbersMapKeys.length; index++) {
    const currentKey = numbersMapKeys[index];
    if (notificationTitle.indexOf(currentKey) > -1) {
      return numbersMap[currentKey];
    }
  }
 
  // Check for specified connector
  for (let index = 0; index < connectorsMapKeys.length; index++) {
    const currentKey = connectorsMapKeys[index];
    if (notificationTitle.indexOf(currentKey) > -1) {
      return connectorsMap[currentKey];
    }
  }
 
  // Fall back to the original title
  return notificationTitle;
}
